 82:      * Ordering
 83:      *
 84:      * Construct the ORDER BY clause for server-side processing SQL query
 85:      *
 86:      *  @param  array $request Data sent to server by DataTables
 87:      *  @param  array $columns Column information array
 88:      *  @return string SQL order by clause
 89:      */
 90:     static function order ( $request, $columns )
 91:     {
 92:         $order = '';
 93: 
 94:         if ( isset($request['order']) && count($request['order']) ) {
 95:             $orderBy = array();
 96:             $dtColumns = self::pluck( $columns, 'dt' );
 97: 
 98:             for ( $i=0, $ien=count($request['order']) ; $i<$ien ; $i++ ) {
 99:                 // Convert the column index into the column data property
100:                 $columnIdx = intval($request['order'][$i]['column']);
101:                 $requestColumn = $request['columns'][$columnIdx];
102: 
103:                 $columnIdx = array_search( $requestColumn['data'], $dtColumns );
104:                 $column = $columns[ $columnIdx ];
105: 
106:                 if ( $requestColumn['orderable'] == 'true' ) {
107:                     $dir = $request['order'][$i]['dir'] === 'asc' ?
108:                         'ASC' :
109:                         'DESC';
110: 
111:                     $orderBy[] = '`'.$column['db'].'` '.$dir;
112:                 }
113:             }
114: 
115:             $order = 'ORDER BY '.implode(', ', $orderBy);
116:         }
117: 
118:         return $order;

122:     /**
123:      * Searching / Filtering
124:      *
125:      * Construct the WHERE clause for server-side processing SQL query.
126:      *
127:      * NOTE this does not match the built-in DataTables filtering which does it
128:      * word by word on any field. It's possible to do here performance on large
129:      * databases would be very poor
130:      *
131:      *  @param  array $request Data sent to server by DataTables
132:      *  @param  array $columns Column information array
133:      *  @param  array $bindings Array of values for PDO bindings, used in the
134:      *    sql_exec() function
135:      *  @return string SQL where clause
136:      */
137:     static function filter ( $request, $columns, &$bindings )
138:     {
139:         $globalSearch = array();
140:         $columnSearch = array();
141:         $dtColumns = self::pluck( $columns, 'dt' );
142: 
143:         if ( isset($request['search']) && $request['search']['value'] != '' ) {
144:             $str = $request['search']['value'];
145: 
146:             for ( $i=0, $ien=count($request['columns']) ; $i<$ien ; $i++ ) {
147:                 $requestColumn = $request['columns'][$i];
148:                 $columnIdx = array_search( $requestColumn['data'], $dtColumns );
149:                 $column = $columns[ $columnIdx ];
150: 
151:                 if ( $requestColumn['searchable'] == 'true' ) {
152:                     $binding = self::bind( $bindings, '%'.$str.'%', PDO::PARAM_STR );
153:                     $globalSearch[] = "`".$column['db']."` LIKE ".$binding;
154:                 }
155:             }
156:         }
157: 
158:         // Individual column filtering
159:         for ( $i=0, $ien=count($request['columns']) ; $i<$ien ; $i++ ) {
160:             $requestColumn = $request['columns'][$i];
161:             $columnIdx = array_search( $requestColumn['data'], $dtColumns );
162:             $column = $columns[ $columnIdx ];
163: 
164:             $str = $requestColumn['search']['value'];
165: 
166:             if ( $requestColumn['searchable'] == 'true' &&
167:              $str != '' ) {
168:                 $binding = self::bind( $bindings, '%'.$str.'%', PDO::PARAM_STR );
169:                 $columnSearch[] = "`".$column['db']."` LIKE ".$binding;
170:             }
171:         }
172: 
173:         // Combine the filters into a single string
174:         $where = '';
175: 
176:         if ( count( $globalSearch ) ) {
177:             $where = '('.implode(' OR ', $globalSearch).')';
178:         }
179: 
180:         if ( count( $columnSearch ) ) {
181:             $where = $where === '' ?
182:                 implode(' AND ', $columnSearch) :
183:                 $where .' AND '. implode(' AND ', $columnSearch);
184:         }
185: 
186:         if ( $where !== '' ) {
187:             $where = 'WHERE '.$where;
188:         }
189: 
190:         return $where;
191:     }

194:     /**
195:      * Perform the SQL queries needed for an server-side processing requested,
196:      * utilising the helper functions of this class, limit(), order() and
197:      * filter() among others. The returned array is ready to be encoded as JSON
198:      * in response to an SSP request, or can be modified if needed before
199:      * sending back to the client.
200:      *
201:      *  @param  array $request Data sent to server by DataTables
202:      *  @param  array $sql_details SQL connection details - see sql_connect()
203:      *  @param  string $table SQL table to query
204:      *  @param  string $primaryKey Primary key of the table
205:      *  @param  array $columns Column information array
206:      *  @return array          Server-side processing response array
207:      */
208:     static function simple ( $request, $sql_details, $table, $primaryKey, $columns )
209:     {
210:         $bindings = array();
211:         $db = self::sql_connect( $sql_details );
212: 
213:         // Build the SQL query string from the request
214:         $limit = self::limit( $request, $columns );
215:         $order = self::order( $request, $columns );
216:         $where = self::filter( $request, $columns, $bindings );
217: 
218:         // Main query to actually get the data
219:         $data = self::sql_exec( $db, $bindings,
220:             "SELECT SQL_CALC_FOUND_ROWS `".implode("`, `", self::pluck($columns, 'db'))."`
221:              FROM `$table`
222:              $where
223:              $order
224:              $limit"
225:         );
226: 
227:         // Data set length after filtering
228:         $resFilterLength = self::sql_exec( $db,
229:             "SELECT FOUND_ROWS()"
230:         );
231:         $recordsFiltered = $resFilterLength[0][0];
232: 
233:         // Total data set length
234:         $resTotalLength = self::sql_exec( $db,
235:             "SELECT COUNT(`{$primaryKey}`)
236:              FROM   `$table`"
237:         );
238:         $recordsTotal = $resTotalLength[0][0];
239: 
240: 
241:         /*
242:          * Output
243:          */
244:         return array(
245:             "draw"            => intval( $request['draw'] ),
246:             "recordsTotal"    => intval( $recordsTotal ),
247:             "recordsFiltered" => intval( $recordsFiltered ),
248:             "data"            => self::data_output( $columns, $data )
249:         );
250:     }

285:     /**
286:      * Execute an SQL query on the database
287:      *
288:      * @param  resource $db  Database handler
289:      * @param  array    $bindings Array of PDO binding values from bind() to be
290:      *   used for safely escaping strings. Note that this can be given as the
291:      *   SQL query string if no bindings are required.
292:      * @param  string   $sql SQL query to execute.
293:      * @return array         Result from the query (all rows)
294:      */
295:     static function sql_exec ( $db, $bindings, $sql=null )
296:     {
297:         // Argument shifting
298:         if ( $sql === null ) {
299:             $sql = $bindings;
300:         }
301: 
302:         $stmt = $db->prepare( $sql );
303:         //echo $sql;
304: 
305:         // Bind parameters
306:         if ( is_array( $bindings ) ) {
307:             for ( $i=0, $ien=count($bindings) ; $i<$ien ; $i++ ) {
308:                 $binding = $bindings[$i];
309:                 $stmt->bindValue( $binding['key'], $binding['val'], $binding['type'] );
310:             }
311:         }
312: 
313:         // Execute
314:         try {
315:             $stmt->execute();
316:         }
317:         catch (PDOException $e) {
318:             self::fatal( "An SQL error occurred: ".$e->getMessage() );
319:         }
320: 
321:         // Return all
322:         return $stmt->fetchAll();
323:     }
